# Simplified Collatz Encryption Logic
def collatz_encrypt(data: str, key: int) -> bytes:
    encrypted = []
    for char in data:
        encrypted.append(ord(char) ^ key)
        key = (key * 3 + 1) if (key % 2) else (key // 2)
